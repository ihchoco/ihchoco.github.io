}

function tabMenu($menuItems){

}

//n번째 탭메뉴 선택
function selectMenuItemAt($menuItems, index){
    $menuItems.removeClass("select");
    $menuItems.eq(index).addClass("select");
}

    $menuItems.removeClass("select");
}
